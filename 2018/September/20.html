关于图像资源标签使用

<!DOCTYPE html>
<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8" />
        <title></title>
    </head>
    <body>
        <!--
            audio         使用音频的标签
            src            引入的资源有一定的规律
            controls     控制面板（在不同浏览器下，样式有所不同）
            autoplay     自动播放 
            preload     预加载
            muted        静音
            volume        音量(js控制)
        -->

        <video  controls="controls" preload>
            <source src="media/video.webm" type="video/webm">

            </source>
        </video>

        <!--
            格式工厂等压缩工具的默认配置未必适合浏览器使用
            视频编码        H.264
            音频编码        AAC
        -->

        <audio>
            <source>

            <source/>
        </audio>
        <p></p>
        <!--
            ifram
                可以引用其他页面来使用
                例如：地图组件、客服系统等第三方工具
        -->
        <iframe src="二级.html" width="422px" height="300px"></iframe>

        <!--
            id         必须是唯一的
            class    类（可以重复使用）
            name    名称（input 中表示参数名）
            data-    表示自定义属性 一般会用data-开头

        -->

    </body>
</html>
