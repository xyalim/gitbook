<!-- 这个注释的快捷键是 "ctrl+/" -->
<!DOCTYPE html>
<html>
    <head>
        <!--
            head 
            将        声明
                    关键词
                    描述
                    特殊设置
                    文件应用

            head里面写的内容都不会呈现在页面当中

        -->

        <link rel="stylesheet" href="css/index.css">

        </style>

        <meta charset="utf-8" />

        <title></title>

    </head>
    <body>
        <!--
            语义化

            1.配合seo的优化
            2.方便开发人员理解
            3.表达出预警，与运行环境有关
        -->

        <!--标题 块级元素-->
        <h1>h1</h1>
        <h2>h2</h2>

        <p>段落 块级元素</p>
        <!--独开一行 块级元素-->

        <span>span 行内元素</span>
        <!--span 行内元素 一般用于段落内 共享同一行-->

        <!--
            块级元素
                自占一行 可设置宽高

            行内元素
                随上个元素同行布局 不可设宽高 面积随内容变动

            置换元素
                又能设置宽高 又不占一行 ：img 
                                        vedio      视频
                                        audio      音频
                                        input      输入框
                                        textarea 文本域

        -->
        <!--上标-->
        2<sup>2</sup>


        <!--下标-->
        H<sub>2</sub>O


        <!--自定义标签-->
        <bluej></bluej>

        <br />
        <a href="#xxx">锚点</a>
        <a name="xxx">锚点对应点</a>

        <!--访问线上地址 协议：//二级域名.一级域名-->
        <a href="http://news.baidu.com">跳转百度新闻</a>

        <a href="http://www.baidu.com" target="_top">aaa</a> <!--新标签的打开方式 target-->

        <a href="#down">跳到下面去</a>
        <div id="up">
            哈哈 在上面了
        </div>
        <div class="heightDiv">

        </div>

        <div id="down">
            哈哈 在下面了
        </div>
        <a href="#up">跳到上面去</a>



        <!--src ： souce 表示引入资源文件
            alt ： 图片加载不了时的替代文本
            width：宽度
            height：高度
            一般情况下，布局时都是设置图片的宽度从而保持图片的宽高比



        -->

        <!--
            无序列表

            emmet 语法
            例：

                ul>li*5>p{新闻$} 再按下tab键

            其中 
                >子元素
                *复制多少个相同的元素
                {}标签里的内容
                $自增的数字
        -->


            <!--
                ul 
                    list-style: url(img/drag.png)    使用图片作为列表的样式
                              : none                   不使用列表样式

            -->

            <ul type="circle">
                <li>
                    <p>新闻1</p>
                </li>
                <li>
                    <p>新闻2</p>
                </li>
                <li>
                    <p>新闻3</p>
                </li>
                <li>
                    <p>新闻4</p>
                </li>
                <li>
                    <p>新闻5</p>
                </li>
            </ul>


        <!--自定义列表
            dl dt dd

            dl 定义列表 definition list
            dt 定义标题 definition title
            dd 定义描述 definition description

            可互相嵌套
        -->



    </body>
</html>
