属性选择器

<!DOCTYPE html>
<html>
    <head>
        <meta charset="UTF-8">
        <title>
            属性选择器
        </title>

        <style type="text/css">

            input{
                width: 200px;

            }
            /*属性选择器  Element[att = "value"]
                        标签对象[属性名="值"]     

            */
            input[type="haha"]{
                border-color: red;
            }
            /*表示只要包含该属性就选中*/
            input[name]{
                border-color: blue;
            }

            /*表示匹配 属性的值以空格隔开的片段*/
            input[my~="hello"]{
                border-color: green;
            }


            /*表示匹配以XX为开头的值*/
            input[data^="ab"]
            {
                background-color: yellow;
            }

            /*表示匹配以XX为结尾的值*/
            input[data$="ef"]
            {
                border-color: red;
            }

            /*表示匹配包含XXX的值*/
            input[data*="e"]
            {
                border-color: blue;
            }

            /*表示以XXX为开头的并且以-为连接额值*/
            input[data|="abc"]
            {
                height: 100px;
            }

        </style>

    </head>
    <body>
        <input type="haha"  />
        <input type="password" name />


        <input type="text" my="hello world" />
        <input type="text"  data="abc-def" />


    </body>
</html>

伪类选择器

<!DOCTYPE html>
<html>
    <head>
        <meta charset="UTF-8">
        <title>伪类选择器</title>
        <style type="text/css">

            /*link         默认未访问的状态
              visited     已访问
              hover:     鼠标悬停
              active:     表示点击时的激活状态
            */

            a:link{
                color: gray;
            }
            a:visited{
                color: blue;
            }

            a:hover{
                color: red;
                font-size: 100px;
            }

            a:active{
                color: yellow;
            }

            /*聚焦状态*/
            input:focus{
                /*描边*/
                outline-color: red;

            }

            /*选中状态*/
            /*新 小知识点！！！*/
            input:checked+label{
                color:red;
            }

            ul li:first-child{
                color: red;
            }

            ul li:last-child{
                color: green;
            }

            ul li:nth-child(2){
                color: blue;
            }

            ul li:nth-last-child(2){
                color: skyblue;
            }


            /*新知识点  2的倍数个 （偶数项）*/
            ul li:nth-child(2n){
                color: blue;
            }

            /*新知识点  奇数项*/
            ul li:nth-child(2n-1){
                color: red;
            }

            /*新知识点 除了某个元素不变其他都改变 */
            ul li:not(:nth-child(3)){
                color: blue;
            }


            /*
                 伪类中的child系列
                 nth里面的括号要传入从1开始的值 可以结合未知数n来使用

             */



            ol li:first-of-type{
                color: red;
            }

            ol li:last-of-type{
                color: red;
            }

            ol li:nth-of-type(2){
                color: red;
            }

            /*唯一的一个儿子*/
            ol div:only-of-type{
                color: orange;
            }

            /*    clear:  第n个儿子和同样的类型才能匹配
                 type:     同类型的第n个
            */



        </style>

    </head>
    <body>

        <!--  类class  -->

        <!-- 伪类：
            1.交互的效果
            2.选择第N个元素

        -->

关系选择器


        关系选择器

            A B            后代选择器(A下面的所有B)
            A>B            子代~(A的B子元素)
            A+B            相邻~(A后的[同级]一个B元素)
            A~B            兄弟~(A后的[同级]全部B元素)
